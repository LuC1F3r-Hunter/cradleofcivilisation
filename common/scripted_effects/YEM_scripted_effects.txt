#Add to scripted triggers if more levels of hyper inflation is added
YEM_debt_clear = {
	remove_ideas = {
		YEM_debt_1
		YEM_debt_2
		YEM_debt_3
		YEM_debt_4
		YEM_debt_5
	}
}

#Important to put HIGHEST first
YEM_debt_level_up = {
	# level 5 not relevant. Cannot go above.
	custom_effect_tooltip = YEM_debt_up_tt
	if = {
		limit = { has_idea = YEM_debt_4 }
		swap_ideas = {
			remove_idea = YEM_debt_4
			add_idea = YEM_debt_5
		}
	}
	else_if = {
		limit = { has_idea = YEM_debt_3 }
		swap_ideas = {
			remove_idea = YEM_debt_3
			add_idea = YEM_debt_4
		}
	}
	else_if = {
		limit = { has_idea = YEM_debt_2 }
		swap_ideas = {
			remove_idea = YEM_debt_2
			add_idea = YEM_debt_3
		}
	}
	else_if = {
		limit = { has_idea = YEM_debt_1 }
		swap_ideas = {
			remove_idea = YEM_debt_1
			add_idea = YEM_debt_2
		}		
	}
	else_if = {
		limit = { NOT = { has_idea = YEM_debt_1 } }
		add_ideas = YEM_debt_1
	}
}

#Important to put LOWEST first
YEM_debt_level_down = {
custom_effect_tooltip = YEM_debt_down_tt
	if = {
		limit = { has_idea = YEM_debt_1 }
		#Makes the tooltip clearer if player looks at the effect early in the game
		effect_tooltip = {
			swap_ideas = {
				remove_idea = YEM_debt_1
				add_idea = YEM_debt_0
			}
		}
		hidden_effect = { remove_ideas = YEM_debt_1 }
	}
	if = {
		limit = { has_idea = YEM_debt_2 }
		swap_ideas = {
			remove_idea = YEM_debt_2
			add_idea = YEM_debt_1
		}
	}
	if = {
		limit = { has_idea = YEM_debt_3 }
		swap_ideas = {
			remove_idea = YEM_debt_3
			add_idea = YEM_debt_2
		}
	}
	if = {
		limit = { has_idea = YEM_debt_4 }
		swap_ideas = {
			remove_idea = YEM_debt_4
			add_idea = YEM_debt_3
		}
	}
	if = {
		limit = { has_idea = YEM_debt_5 }
		swap_ideas = {
			remove_idea = YEM_debt_5
			add_idea = YEM_debt_4
		}
	}
}
